/**
 * 全局键盘快捷键 composable。
 * 用法：useKeyboard({ 'Ctrl+S': () => save(), 'ArrowLeft': () => prev() })
 * 输入框聚焦时不拦截单键（仅组合键生效）。
 */
function useKeyboard(shortcuts) {
  function handler(e) {
    const tag = e.target.tagName;
    const isInput = tag === 'INPUT' || tag === 'TEXTAREA' || tag === 'SELECT' || e.target.isContentEditable;

    // 单键（无修饰符）在输入框内不拦截
    const hasMod = e.ctrlKey || e.metaKey || e.altKey;
    if (isInput && !hasMod) return;

    const key = [];
    if (e.ctrlKey || e.metaKey) key.push('Ctrl');
    if (e.altKey) key.push('Alt');
    if (e.shiftKey) key.push('Shift');
    key.push(e.key.length === 1 ? e.key.toUpperCase() : e.key);

    const combo = key.join('+');
    const action = shortcuts[combo] || shortcuts[e.key];
    if (action) {
      e.preventDefault();
      action();
    }
  }

  document.addEventListener('keydown', handler);
  return () => document.removeEventListener('keydown', handler);
}
```

/**
 * SortableJS 封装 composable。
 * 用法：useDragSort(containerEl, items, { onEnd: (items) => save(items) })
 * 返回 cleanup 函数。
 */
function useDragSort(el, items, opts = {}) {
  if (!el || typeof Sortable === 'undefined') return () => {};

  const instance = Sortable.create(el, {
    animation: 150,
    handle: '.drag-handle',
    ghostClass: 'sortable-ghost',
    dragClass: 'sortable-drag',
    onEnd(evt) {
      const moved = items.splice(evt.oldIndex, 1)[0];
      items.splice(evt.newIndex, 0, moved);
      if (opts.onEnd) opts.onEnd(items);
    },
    ...opts,
  });

  return () => instance.destroy();
}
```

const WeekPlan = {
  props: {
    weekData: Object,
  },
  emits: ['save'],
  data() {
    return {
      fields: {
        mainLine: [],      // 本周主线
        mustComplete: [],  // 本周必须完成
        pushForward: [],   // 本周推进即可
        tempIn: [],        // 临时进入
      },
      saving: false,
      saved: false,
    };
  },
  watch: {
    weekData: {
      immediate: true,
      handler(val) {
        if (val && Object.keys(val).length) {
          this.fields.mainLine = split(val['本周主线']);
          this.fields.mustComplete = split(val['本周必须完成']);
          this.fields.pushForward = split(val['本周推进即可']);
          this.fields.tempIn = split(val['临时进入']);
        }
        this.ensureOne();
      },
    },
  },
  methods: {
    split(s) { return (s || '').split('\n').filter(l => l.trim()); },
    join(a) { return a.filter(s => s.trim()).join('\n'); },
    ensureOne() {
      ['mainLine','mustComplete','pushForward','tempIn'].forEach(k => {
        if (!this.fields[k].length) this.fields[k] = [''];
      });
    },
    addItem(field) { this.fields[field].push(''); },
    removeItem(field, i) { this.fields[field].splice(i, 1); this.ensureOne(); this.debouncedSave(); },
    onInput(field, i, e) { this.fields[field][i] = e.target.value; this.debouncedSave(); },
    debouncedSave() {
      clearTimeout(this._st);
      this._st = setTimeout(() => this.doSave(), 500);
    },
    async doSave() {
      this.saving = true;
      const payload = {
        '本周主线': this.join(this.fields.mainLine),
        '本周必须完成': this.join(this.fields.mustComplete),
        '本周推进即可': this.join(this.fields.pushForward),
        '临时进入': this.join(this.fields.tempIn),
      };
      this.$emit('save', payload);
      this.saved = true;
      this.saving = false;
      setTimeout(() => this.saved = false, 2000);
    },
    promoteTempToTasks() {
      this.$emit('promote-temp', this.fields.tempIn.filter(s => s.trim()));
    },
  },
  template: `
    <section class="panel weekly-panel">
      <div class="panel-head">
        <div><p class="panel-kicker">周计划 · 方向</p><h2>先把本周方向定住</h2></div>
        <div style="display:flex;gap:8px;align-items:center;">
          <span v-if="saved" style="color:var(--mint);font-size:12px;font-weight:700;">✓ 已保存</span>
          <button class="primary" @click="doSave" :disabled="saving">{{ saving?'...':'保存' }}</button>
        </div>
      </div>

      <div class="two-col">
        <div v-for="(cfg, key) in [
          {key:'mainLine',label:'本周主线',placeholder:'每条一行。不是任务清单，是方向。'},
          {key:'mustComplete',label:'本周必须完成',placeholder:'每条一行。'},
          {key:'pushForward',label:'本周推进即可',placeholder:'每条一行。'},
          {key:'tempIn',label:'临时进入',placeholder:'新任务缓冲区。'}
        ]" :key="cfg.key" style="display:flex;flex-direction:column;gap:6px;">
          <span style="font-size:13px;font-weight:600;color:var(--muted);">{{ cfg.label }}</span>
          <div v-for="(item, i) in fields[cfg.key]" :key="cfg.key+i" style="display:flex;align-items:center;gap:6px;">
            <span class="drag-handle" style="color:#ccc;cursor:grab;font-size:14px;user-select:none;">⠿</span>
            <input :value="item" @input="onInput(cfg.key, i, $event)" :placeholder="cfg.placeholder"
              style="flex:1;min-height:42px;padding:8px 12px;border-radius:12px;border:1px solid rgba(41,31,18,.12);background:rgba(255,255,255,.82);font-size:14px;" />

          </div>
          <button @click="addItem(cfg.key)" style="width:100%;padding:10px;border:1.5px dashed var(--line);border-radius:14px;background:transparent;color:var(--muted);font-size:13px;cursor:pointer;">+ 添加</button>
        </div>
      </div>

const YesterdayRef = {
  props: {
    wk: String,
    day: String,
  },
  emits: ['inherit'],
  data() {
    return { refText: '', pendingItems: [], loading: true };
  },
  watch: {
    day: { immediate: true, handler: 'load' },
  },
  methods: {
    async load() {
      this.loading = true;
      const d = new Date(this.day);
      d.setDate(d.getDate() - 1);
      const yd = d.toISOString().split('T')[0];
      try {
        const r = await (await fetch('/api/weeks/' + this.wk + '/days/' + yd)).json();
        if (r.code === 0 && r.data) {
          const f = r.data;
          const parts = [];
          this.pendingItems = [];
          if (f['明日优先']) {
            parts.push('明日:' + f['明日优先'].replace(/\n/g, ' | '));
            this.pendingItems.push(...(f['明日优先'] || '').split('\n').filter(s => s.trim()));
          }
          if (f['收尾_迁移']) {
            parts.push('迁移:' + f['收尾_迁移'].replace(/\n/g, ' | '));
            this.pendingItems.push(...(f['收尾_迁移'] || '').split('\n').filter(s => s.trim()));
          }
          this.refText = parts.join(' ； ') || '无记录';
        } else {
          this.refText = '无记录';
          this.pendingItems = [];
        }
      } catch (e) {
        this.refText = '无记录';
      }
      this.loading = false;
    },
  },
  template: `
    <section v-if="refText!=='无记录'&&!loading" class="panel"
      :style="{padding:'16px 20px',marginBottom:'14px',borderLeft: pendingItems.length?'3px solid #f0a020':'3px solid var(--mint)'}">
      <div style="display:flex;align-items:center;gap:10px;flex-wrap:wrap;">
        <span style="font-size:13px;color:var(--muted);flex:1;">💬 昨日参考：{{ refText }}</span>
        <span v-if="pendingItems.length" style="font-size:11px;color:#f0a020;font-weight:600;">{{ pendingItems.length }}条待继承</span>
        <button class="ghost" @click="$emit('inherit', pendingItems)">📥 一键继承到今日</button>
      </div>
    </section>
  `,
};

const TaskPool = {
  props: {
    tasks: Array,
  },
  emits: ['toggle', 'delete', 'add', 'reorder', 'add-to-focus'],
  data() {
    return { showForm: false, newContent: '', newType: '必须完成' };
  },
  computed: {
    progress() {
      const total = this.tasks.length;
      const done = this.tasks.filter(t => t['完成状态']).length;
      return { total, done, rate: total ? Math.round(done / total * 100) : 0 };
    },
  },
  mounted() {
    this.$nextTick(() => {
      const el = this.$el.querySelector('.task-list');
      if (el) {
        this._cleanup = useDragSort(el, this.tasks, {
          onEnd: (items) => this.$emit('reorder', items.map(t => t.record_id)),
        });
      }
    });
  },

  methods: {

      if (!this.newContent.trim()) return;

      if (type === '推进即可') return 'type-push';

    startAutoSave() { clearInterval(this.autoSaveTimer); this.autoSaveTimer = setInterval(() => { this.saveDay(); this.onSaveWeek(this.w); }, 60000); },

    /* ── 任务操作 ── */
    async onToggleTask(t) {
      t['完成状态'] = !t['完成状态'];
      try {
        await $.put('/weeks/' + this.wk + '/tasks/' + t.record_id, { '完成状态': t['完成状态'] });
        if (t['完成状态']) {
          this.doneItems.push(t['任务内容']);
          this.saveDay();
          this.toast = '✅ 已标记完成'; setTimeout(() => this.toast = '', 1500);
        }
      } catch (e) { t['完成状态'] = !t['完成状态']; }
    },
    async onDeleteTask(id) { await $.del('/weeks/' + this.wk + '/tasks/' + id); await this.loadTasks(); },
    async onAddTask({ content, task_type }) {
      await $.post('/weeks/' + this.wk + '/tasks', { content, task_type });
      await this.loadTasks();
    },
    async onReorderTasks(ids) {
      await $.put('/weeks/' + this.wk + '/tasks/reorder', ids);
    },
    onAddTaskToFocus(text) {
      if (!this.focus.includes(text)) {
        if (this.focus[0] === '') this.focus[0] = text;
        else this.focus.push(text);
        this.saveDay();
      }
    },

    /* ── 今日重点操作 ── */
    onMoveFocus({ item, target }) {
      if (target === 'done') { this.doneItems.push(item); this.toast = '✅ 已归入完成区'; }
      else if (target === 'migrated') { this.migItems.push(item); this.toast = '➡️ 已归入迁移区'; }
      this.saveDay();
      setTimeout(() => this.toast = '', 1500);
    },
    onReorderFocus() { this.saveDay(); },

    /* ── 收尾 ── */
    async onQuickClose() {
      const remain = this.focus.filter(s => s.trim());
      if (remain.length) { this.doneItems.push(...remain); this.focus = ['']; }
      await this.saveDay();
      try {
        await $.post('/weeks/' + this.wk + '/days/' + this.day + '/close');
        this.closed = true;
        this.dayStatuses[this.day] = 'closed';
        this.toast = '🎉 收尾完成！'; setTimeout(() => this.toast = '', 2000);
      } catch (e) {}
    },

    /* ── 昨日继承 ── */
    async onInheritYesterday(items) {
      if (!items || !items.length) {
        // 无传入 items 时自动从昨日获取
        const d = new Date(this.day); d.setDate(d.getDate() - 1);
        const yd = d.toISOString().split('T')[0];
        try {
          const r = await $.get('/weeks/' + this.wk + '/days/' + yd);
          if (r.code === 0 && r.data) {
            const f = r.data;
            items = [];
            if (f['明日优先']) items.push(...(f['明日优先'] || '').split('\n').filter(s => s.trim()));
            if (f['收尾_迁移']) items.push(...(f['收尾_迁移'] || '').split('\n').filter(s => s.trim()));
          }
        } catch (e) { return; }
      }
      for (const item of items) {
        if (item.trim() && !this.focus.includes(item)) {
          if (this.focus[0] === '') this.focus[0] = item;
          else this.focus.push(item);
        }
      }
      if (this.focus[0] === '') this.focus = [''];
      this.saveDay();
      this.toast = '📥 已继承昨日'; setTimeout(() => this.toast = '', 1500);
    },

    /* ── 临时进入 → 任务 ── */
    async onPromoteTemp(lines) {
      for (const l of lines) {
        try { await $.post('/weeks/' + this.wk + '/tasks', { content: l, task_type: '临时进入' }); } catch (e) {}
      }
      await this.loadTasks();
      this.toast = '📦 已批量创建为周任务'; setTimeout(() => this.toast = '', 1500);
    },

    /* ── 周复盘 ── */
    async genReview() {
      try {
        const [dr, wr] = await Promise.all([$.get('/weeks/' + this.wk + '/days'), $.get('/weeks/' + this.wk)]);
        const days = dr.code === 0 ? (dr.data || []) : [];
        const L = [];
        const week = wr.code === 0 ? (wr.data || {}) : {};
        if (week['本周主线']) L.push('## 本周主线\n' + week['本周主线']);
        L.push('## 每日完成');
        for (const d of days) { if (d['收尾_完成']?.trim()) L.push(d['星期'] + '：' + d['收尾_完成'].trim().replace(/\n/g, '；')); }
        const ints = days.filter(d => d['主要打断']?.trim()).map(d => d['星期'] + '：' + d['主要打断'].trim().replace(/\n/g, '；'));
        if (ints.length) L.push('## 风险与打断\n' + ints.join('\n'));
        this.reviewText = L.join('\n\n');
      } catch (e) {}
    },
    onSaveReview(text) { $.put('/weeks/' + this.wk, { '本周总结': text }); },
    exportWeek() { window.open('/api/weeks/' + this.wk + '/export', '_blank'); },
  },
});

app.component('WeekNav', WeekNav);
app.component('WeekPlan', WeekPlan);
app.component('DayTabs', DayTabs);
app.component('YesterdayRef', YesterdayRef);
app.component('TaskPool', TaskPool);
app.component('DailyFocus', DailyFocus);
app.component('DailyReview', DailyReview);
app.component('WeekReview', WeekReview);

app.mount('#app');
</script>
</body>
</html>
